# rocky linux packer 설치
sudo yum install -y yum-utils
sudo yum-config-manager --add-repo https://rpm.releases.hashicorp.com/RHEL/hashicorp.repo
sudo yum -y install packer

# 아래의 동작이 잘 되기 위해서는 aws 설정이 되어 있어야 한다 (인증키)
# ansible을 사용할거라면 ansible 도 설치가 되어 있어야한다. (mgmt) 서버에
# 아래의 작업은 설정만 미리 되어있으면 깃허브 액션스에서도 사용이 가능하다

# packer 초기화
packer init ami_was.pkr.hcl

# packer 로 ami 이미지만들어서 aws에 등록
packer build ami_was.pkr.hcl
